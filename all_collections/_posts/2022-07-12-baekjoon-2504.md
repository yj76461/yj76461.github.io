#2504번 괄호의 값 문제 풀이

``` c++

#include <stdio.h>
#include <iostream>
#include <stack>

using namespace std;

char arr[50] = {0, };
int length = 0;
stack<char> s1;

int calc(char arr[], int idx){
    int tmp = 1;
    int res = 0;
    int a = 0, b = 0; // 괄호 개수 카운트 변수

    for(int i = 0; i < idx ; i++){
        s1.push(arr[i]);
        if(s1.top() == '('){
            tmp *= 2;
            a++;
        }
        else if(s1.top() == '['){
            tmp *= 3;
            b++;
        }
        else if(s1.top() == ')'){
            s1.pop();
            a--;

            if(s1.empty() || s1.top() == '[') // 닫히는 괄호로 시작했거나 잘못된 짝이 있는 경우를 잡아내기 위함.
                return 0;

            else if(s1.top() == '('){ // 직전의 괄호가 여는 괄호면 결과값에 현재 값을 더하고 2로 나눈다.
                s1.push(arr[i]);
                res += tmp;
                tmp /= 2; // 2로 나누는 것은 현재 괄호를 탈출하면서 다른 괄호와의 합 연산을 준비하기 위함.
            }
            else{   // 닫는 괄호라면 별다른 연산없이 2로 나눈다.
                s1.push(arr[i]);
                tmp /= 2;
            }
        }
        else if(s1.top() == ']'){
            s1.pop();
            b--;

            if(s1.empty() || s1.top() == '(') // 닫히는 괄호로 시작했거나 잘못된 짝이 있는 경우를 잡아내기 위함.
                return 0;

            else if(s1.top() == '['){ // 직전의 괄호가 여는 괄호면 결과값에 현재 값을 더하고 3으로 나눈다.
                s1.push(arr[i]);
                res += tmp;
                tmp /= 3; // 3으로 나누는 것은 현재 괄호를 탈출하면서 다른 괄호와의 합 연산을 준비하기 위함.
            }
            else{   // 닫는 괄호라면 별다른 연산없이 3으로 나눈다.
                s1.push(arr[i]);
                tmp /= 3;
            }
        }
        // 검산하기 위해 작성한 코드
        //cout << " tmp: " << tmp << " res: " << res << " empty: " << s1.empty() << endl;
    }

    // 괄호의 개수와 짝이 맞는 지 검사.
    if(!(a == 0 && b == 0))
        res = 0;

    return res;
}

int main(){
    scanf("%s", arr);
    while(arr[length] != '\0')
        length++;
    
    cout << calc(arr, length) << endl;
    return 0;
}
```
